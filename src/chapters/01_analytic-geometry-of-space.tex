\chapter{空间解析几何}

\section{向量与坐标系}

\begin{problem}{向量数乘的运算性质}{Vector-Scalar-Multiplication-Properties}
    设 $\lambda, \mu \in \symbb{R}$，$\symbfit{a}, \symbfit{b}$ 为向量，证明如下性质：
    \begin{enumerate}
        \item $\lambda(\symbfit{a} + \symbfit{b}) = \lambda\symbfit{a} + \lambda\symbfit{b}$
        \item $(\lambda + \mu)\symbfit{a} = \lambda\symbfit{a} + \mu\symbfit{a}$
        \item $(\lambda\mu)\symbfit{a} = \lambda(\mu\symbfit{a})$
    \end{enumerate}
\end{problem}

\begin{myproof}
    设 $\symbfit{a} = (a_1, a_2, a_3)$，$\symbfit{b} = (b_1, b_2, b_3)$。

    \ansitem{1}

    \begin{equation*}
        \begin{aligned}
            \lambda(\symbfit{a} + \symbfit{b}) & = \lambda(a_1 + b_1, a_2 + b_2, a_3 + b_3)                                          \\
                                               & = (\lambda(a_1 + b_1), \lambda(a_2 + b_2), \lambda(a_3 + b_3))                      \\
                                               & = (\lambda a_1 + \lambda b_1, \lambda a_2 + \lambda b_2, \lambda a_3 + \lambda b_3) \\
                                               & = (\lambda a_1, \lambda a_2, \lambda a_3) + (\lambda b_1, \lambda b_2, \lambda b_3) \\
                                               & = \lambda\symbfit{a} + \lambda\symbfit{b}
        \end{aligned}
    \end{equation*}
    故得证：
    \begin{equation*}
        \boxed{\lambda(\symbfit{a} + \symbfit{b}) = \lambda\symbfit{a} + \lambda\symbfit{b}}
    \end{equation*}

    \ansitem{2}

    \begin{equation*}
        \begin{aligned}
            (\lambda + \mu)\symbfit{a} & = ((\lambda + \mu)a_1, (\lambda + \mu)a_2, (\lambda + \mu)a_3)          \\
                                       & = (\lambda a_1 + \mu a_1, \lambda a_2 + \mu a_2, \lambda a_3 + \mu a_3) \\
                                       & = (\lambda a_1, \lambda a_2, \lambda a_3) + (\mu a_1, \mu a_2, \mu a_3) \\
                                       & = \lambda\symbfit{a} + \mu\symbfit{a}
        \end{aligned}
    \end{equation*}
    故得证：
    \begin{equation*}
        \boxed{(\lambda + \mu)\symbfit{a} = \lambda\symbfit{a} + \mu\symbfit{a}}
    \end{equation*}

    \ansitem{3}

    \begin{equation*}
        \begin{aligned}
            (\lambda\mu)\symbfit{a} & = ((\lambda\mu)a_1, (\lambda\mu)a_2, (\lambda\mu)a_3)    \\
                                    & = (\lambda(\mu a_1), \lambda(\mu a_2), \lambda(\mu a_3)) \\
                                    & = \lambda(\mu a_1, \mu a_2, \mu a_3)                     \\
                                    & = \lambda(\mu\symbfit{a})
        \end{aligned}
    \end{equation*}
    故得证：
    \begin{equation*}
        \boxed{(\lambda\mu)\symbfit{a} = \lambda(\mu\symbfit{a})}
    \end{equation*}
\end{myproof}


\begin{problem}{向量积的运算性质}{Vector-Cross-Product-Properties}
    证明向量积的如下性质：
    \begin{enumerate}
        \item $(\lambda\symbfit{a}) \times \symbfit{b} = \symbfit{a} \times (\lambda\symbfit{b}) = \lambda(\symbfit{a} \times \symbfit{b})$，其中 $\lambda \in \symbb{R}$。
        \item $\symbfit{a} \times \symbfit{b} = -\symbfit{b} \times \symbfit{a}$
    \end{enumerate}
\end{problem}

\begin{myproof}
    设 $\symbfit{a} = (a_1, a_2, a_3)$，$\symbfit{b} = (b_1, b_2, b_3)$，基向量为 $\symbfit{i}, \symbfit{j}, \symbfit{k}$。

    \ansitem{1}

    根据向量积的行列式定义与行列式性质：
    \begin{equation*}
        \begin{aligned}
            (\lambda\symbfit{a}) \times \symbfit{b} & = \begin{vmatrix} \symbfit{i} & \symbfit{j} & \symbfit{k} \\ \lambda a_1 & \lambda a_2 & \lambda a_3 \\ b_1 & b_2 & b_3 \end{vmatrix} \\
                                                    & = \lambda \begin{vmatrix} \symbfit{i} & \symbfit{j} & \symbfit{k} \\ a_1 & a_2 & a_3 \\ b_1 & b_2 & b_3 \end{vmatrix}                 \\
                                                    & = \lambda(\symbfit{a} \times \symbfit{b})
        \end{aligned}
    \end{equation*}
    同理可得：
    \begin{equation*}
        \begin{aligned}
            \symbfit{a} \times (\lambda\symbfit{b}) & = \begin{vmatrix} \symbfit{i} & \symbfit{j} & \symbfit{k} \\ a_1 & a_2 & a_3 \\ \lambda b_1 & \lambda b_2 & \lambda b_3 \end{vmatrix} \\
                                                    & = \lambda \begin{vmatrix} \symbfit{i} & \symbfit{j} & \symbfit{k} \\ a_1 & a_2 & a_3 \\ b_1 & b_2 & b_3 \end{vmatrix}                 \\
                                                    & = \lambda(\symbfit{a} \times \symbfit{b})
        \end{aligned}
    \end{equation*}
    由此得到：
    \begin{equation*}
        \boxed{(\lambda\symbfit{a}) \times \symbfit{b} = \symbfit{a} \times (\lambda\symbfit{b}) = \lambda(\symbfit{a} \times \symbfit{b})}
    \end{equation*}

    \ansitem{2}

    根据行列式交换两行改变符号的性质：
    \begin{equation*}
        \begin{aligned}
            \symbfit{a} \times \symbfit{b} & = \begin{vmatrix} \symbfit{i} & \symbfit{j} & \symbfit{k} \\ a_1 & a_2 & a_3 \\ b_1 & b_2 & b_3 \end{vmatrix}   \\
                                           & = - \begin{vmatrix} \symbfit{i} & \symbfit{j} & \symbfit{k} \\ b_1 & b_2 & b_3 \\ a_1 & a_2 & a_3 \end{vmatrix} \\
                                           & = -\symbfit{b} \times \symbfit{a}
        \end{aligned}
    \end{equation*}
    即有：
    \begin{equation*}
        \boxed{\symbfit{a} \times \symbfit{b} = -\symbfit{b} \times \symbfit{a}}
    \end{equation*}
\end{myproof}


\begin{problem}{向量代数基本性质}{Vector-Algebra-Properties}
    判断下列结论是否成立，并举例说明：
    \begin{enumerate}
        \item 若 $\symbfit{a} \cdot \symbfit{b} = 0$, 则 $\symbfit{a} = \symbfit{0}$ 或 $\symbfit{b} = \symbfit{0}$.
        \item 若 $\symbfit{a} \times \symbfit{c} = \symbfit{a} \times \symbfit{b}$, 则必有 $\symbfit{c} = \symbfit{b}$.
        \item 两单位向量的数量积必等于 $1$, 向量积必等于一单位向量.
        \item $(\symbfit{a} \cdot \symbfit{b})\symbfit{c} = \symbfit{a}(\symbfit{b} \cdot \symbfit{c})$.
        \item $|\symbfit{a} \cdot \symbfit{b}|^2 = |\symbfit{a}|^2 \cdot |\symbfit{b}|^2$.
        \item $(\symbfit{a} + \symbfit{b}) \times (\symbfit{a} + \symbfit{b}) = \symbfit{a} \times \symbfit{a} + 2\symbfit{a} \times \symbfit{b} + \symbfit{b} \times \symbfit{b}$.
    \end{enumerate}
\end{problem}

\begin{solution}
    \ansitem{1} 不成立。

    两个非零向量垂直时，其数量积为零。设 $\symbfit{a} = (1, 0, 0)$，$\symbfit{b} = (0, 1, 0)$，则有
    \begin{equation*}
        \symbfit{a} \cdot \symbfit{b} = 1 \cdot 0 + 0 \cdot 1 + 0 \cdot 0 = 0
    \end{equation*}
    显然 $\symbfit{a} \neq \symbfit{0}$ 且 $\symbfit{b} \neq \symbfit{0}$。
    \begin{equation*}
        \boxed{\text{结论不成立}}
    \end{equation*}

    \ansitem{2} 不成立。

    由 $\symbfit{a} \times \symbfit{c} = \symbfit{a} \times \symbfit{b} \implies \symbfit{a} \times (\symbfit{c} - \symbfit{b}) = \symbfit{0}$，这仅说明 $\symbfit{a}$ 与 $\symbfit{c} - \symbfit{b}$ 共线。
    取 $\symbfit{a} = (1, 0, 0)$，$\symbfit{b} = (0, 1, 0)$，$\symbfit{c} = (1, 1, 0)$，则
    \begin{equation*}
        \begin{aligned}
            \symbfit{a} \times \symbfit{b} & = (0, 0, 1)                                                                                                   \\
            \symbfit{a} \times \symbfit{c} & = \begin{vmatrix} \symbfit{i} & \symbfit{j} & \symbfit{k} \\ 1 & 0 & 0 \\ 1 & 1 & 0 \end{vmatrix} = (0, 0, 1)
        \end{aligned}
    \end{equation*}
    虽然 $\symbfit{a} \times \symbfit{c} = \symbfit{a} \times \symbfit{b}$，但 $\symbfit{c} \neq \symbfit{b}$。
    \begin{equation*}
        \boxed{\text{结论不成立}}
    \end{equation*}

    \ansitem{3} 不成立。

    设两单位向量为 $\symbfit{e}_1, \symbfit{e}_2$，夹角为 $\theta$。
    其数量积 $\symbfit{e}_1 \cdot \symbfit{e}_2 = |\symbfit{e}_1||\symbfit{e}_2|\cos\theta = \cos\theta$，仅在 $\theta = 0$ 时等于 $1$。
    其向量积模长 $|\symbfit{e}_1 \times \symbfit{e}_2| = |\symbfit{e}_1||\symbfit{e}_2|\sin\theta = \sin\theta$，仅在 $\theta = \frac{\uppi}{2}$ 时为单位向量。
    取 $\symbfit{e}_1 = (1, 0, 0)$，$\symbfit{e}_2 = (0, 1, 0)$，则 $\symbfit{e}_1 \cdot \symbfit{e}_2 = 0 \neq 1$；
    取 $\symbfit{e}_1 = \symbfit{e}_2$，则 $\symbfit{e}_1 \times \symbfit{e}_2 = \symbfit{0}$，非单位向量。
    \begin{equation*}
        \boxed{\text{结论不成立}}
    \end{equation*}

    \ansitem{4} 不成立。

    $(\symbfit{a} \cdot \symbfit{b})\symbfit{c}$ 的方向与 $\symbfit{c}$ 相同或相反，而 $\symbfit{a}(\symbfit{b} \cdot \symbfit{c})$ 的方向与 $\symbfit{a}$ 相同或相反。
    设 $\symbfit{a} = (1, 0, 0)$，$\symbfit{b} = (1, 0, 0)$，$\symbfit{c} = (0, 1, 0)$，则
    \begin{equation*}
        \begin{aligned}
            (\symbfit{a} \cdot \symbfit{b})\symbfit{c} & = 1 \cdot (0, 1, 0) = (0, 1, 0) \\
            \symbfit{a}(\symbfit{b} \cdot \symbfit{c}) & = (1, 0, 0) \cdot 0 = (0, 0, 0)
        \end{aligned}
    \end{equation*}
    二者不相等。
    \begin{equation*}
        \boxed{\text{结论不成立}}
    \end{equation*}

    \ansitem{5} 不成立。

    根据定义 $|\symbfit{a} \cdot \symbfit{b}|^2 = |\symbfit{a}|^2 |\symbfit{b}|^2 \cos^2\theta$。只有当 $\cos^2\theta = 1$（即两向量共线）时才成立。
    取 $\symbfit{a} = (1, 0, 0)$，$\symbfit{b} = (0, 1, 0)$，则
    \begin{equation*}
        \begin{aligned}
            |\symbfit{a} \cdot \symbfit{b}|^2 & = |0|^2 = 0         \\
            |\symbfit{a}|^2 |\symbfit{b}|^2   & = 1^2 \cdot 1^2 = 1
        \end{aligned}
    \end{equation*}
    \begin{equation*}
        \boxed{\text{结论不成立}}
    \end{equation*}

    \ansitem{6} 不成立。

    由向量积的分配律及反交换性可知
    \begin{equation*}
        \begin{aligned}
            (\symbfit{a} + \symbfit{b}) \times (\symbfit{a} + \symbfit{b}) & = \symbfit{a} \times \symbfit{a} + \symbfit{a} \times \symbfit{b} + \symbfit{b} \times \symbfit{a} + \symbfit{b} \times \symbfit{b} \\
                                                                           & = \symbfit{0} + \symbfit{a} \times \symbfit{b} - \symbfit{a} \times \symbfit{b} + \symbfit{0}                                       \\
                                                                           & = \symbfit{0}
        \end{aligned}
    \end{equation*}
    而原命题右端为 $\symbfit{a} \times \symbfit{a} + 2\symbfit{a} \times \symbfit{b} + \symbfit{b} \times \symbfit{b} = 2\symbfit{a} \times \symbfit{b}$。
    取 $\symbfit{a} = (1, 0, 0)$，$\symbfit{b} = (0, 1, 0)$，则右端为 $(0, 0, 2) \neq \symbfit{0}$。
    \begin{equation*}
        \boxed{\text{结论不成立}}
    \end{equation*}
\end{solution}


\begin{problem}{向量混合积的轮换对称性}{Triple-Scalar-Product}
    证明：$\symbfit{a} \times \symbfit{b} \cdot \symbfit{c} = \symbfit{b} \times \symbfit{c} \cdot \symbfit{a} = \symbfit{c} \times \symbfit{a} \cdot \symbfit{b}$.
\end{problem}

\begin{myproof}
    设各向量在直角坐标系下的坐标分别为 $\symbfit{a} = (a_1, a_2, a_3)$，$\symbfit{b} = (b_1, b_2, b_3)$，$\symbfit{c} = (c_1, c_2, c_3)$。根据向量混合积的定义可知，其值等于以这三个向量为棱的平行六面体的体积，对应行列式形式为
    \begin{equation*}
        \symbfit{a} \times \symbfit{b} \cdot \symbfit{c} = \begin{vmatrix} a_1 & a_2 & a_3 \\ b_1 & b_2 & b_3 \\ c_1 & c_2 & c_3 \end{vmatrix}
    \end{equation*}
    根据行列式性质，交换任意两行，行列式的值反号。通过两次行对换（即一次轮换），行列式的值保持不变，则有
    \begin{equation*}
        \begin{aligned}
            \begin{vmatrix} a_1 & a_2 & a_3 \\ b_1 & b_2 & b_3 \\ c_1 & c_2 & c_3 \end{vmatrix}
             & = - \begin{vmatrix} b_1 & b_2 & b_3 \\ a_1 & a_2 & a_3 \\ c_1 & c_2 & c_3 \end{vmatrix}                                                  \\
             & = \begin{vmatrix} b_1 & b_2 & b_3 \\ c_1 & c_2 & c_3 \\ a_1 & a_2 & a_3 \end{vmatrix} = \symbfit{b} \times \symbfit{c} \cdot \symbfit{a}
        \end{aligned}
    \end{equation*}
    同理，对上述行列式再次进行两次行对换可得
    \begin{equation*}
        \begin{vmatrix} b_1 & b_2 & b_3 \\ c_1 & c_2 & c_3 \\ a_1 & a_2 & a_3 \end{vmatrix}
        = - \begin{vmatrix} c_1 & c_2 & c_3 \\ b_1 & b_2 & b_3 \\ a_1 & a_2 & a_3 \end{vmatrix}
        = \begin{vmatrix} c_1 & c_2 & c_3 \\ a_1 & a_2 & a_3 \\ b_1 & b_2 & b_3 \end{vmatrix} = \symbfit{c} \times \symbfit{a} \cdot \symbfit{b}
    \end{equation*}
    综上所述，原命题得证
    \begin{equation*}
        \boxed{\symbfit{a} \times \symbfit{b} \cdot \symbfit{c} = \symbfit{b} \times \symbfit{c} \cdot \symbfit{a} = \symbfit{c} \times \symbfit{a} \cdot \symbfit{b}}
    \end{equation*}
\end{myproof}


\begin{problem}{向量中点公式}{Vector-Midpoint-Formula}
    设 $\overrightarrow{AM} = \overrightarrow{MB}$，证明：对任意一点 $O$，
    \begin{equation*}
        \overrightarrow{OM} = \frac{1}{2} (\overrightarrow{OA} + \overrightarrow{OB}).
    \end{equation*}
\end{problem}

\begin{myproof}
    利用向量的减法分解性质，将已知条件中的向量以点 $O$ 为基准进行拆解
    \begin{equation*}
        \overrightarrow{OM} - \overrightarrow{OA} = \overrightarrow{OB} - \overrightarrow{OM}
    \end{equation*}
    对上式进行移项合并
    \begin{equation*}
        \begin{aligned}
             & \implies \overrightarrow{OM} + \overrightarrow{OM} = \overrightarrow{OA} + \overrightarrow{OB} \\
             & \implies 2 \overrightarrow{OM} = \overrightarrow{OA} + \overrightarrow{OB}
        \end{aligned}
    \end{equation*}
    方程两边同时除以 $2$，即可得到
    \begin{equation*}
        \boxed{\overrightarrow{OM} = \frac{1}{2} (\overrightarrow{OA} + \overrightarrow{OB})}
    \end{equation*}
\end{myproof}


\begin{problem}{向量内积}{Vector-Inner-Product}
    设 $\symbfit{a}, \symbfit{b}, \symbfit{c}$ 是满足 $\symbfit{a} + \symbfit{b} + \symbfit{c} = \symbfit{0}$ 的单位向量，试求 $\symbfit{a} \cdot \symbfit{b} + \symbfit{b} \cdot \symbfit{c} + \symbfit{c} \cdot \symbfit{a}$ 的值.
\end{problem}

\begin{solution}
    由于 $\symbfit{a}, \symbfit{b}, \symbfit{c}$ 为单位向量，故有 $|\symbfit{a}| = |\symbfit{b}| = |\symbfit{c}| = 1$. 对已知等式两边平方得
    \begin{equation*}
        (\symbfit{a} + \symbfit{b} + \symbfit{c})^2 = \symbfit{0}^2 = 0
    \end{equation*}
    展开上式左侧
    \begin{equation*}
        |\symbfit{a}|^2 + |\symbfit{b}|^2 + |\symbfit{c}|^2 + 2(\symbfit{a} \cdot \symbfit{b} + \symbfit{b} \cdot \symbfit{c} + \symbfit{c} \cdot \symbfit{a}) = 0
    \end{equation*}
    代入向量模长
    \begin{equation*}
        1 + 1 + 1 + 2(\symbfit{a} \cdot \symbfit{b} + \symbfit{b} \cdot \symbfit{c} + \symbfit{c} \cdot \symbfit{a}) = 0
    \end{equation*}
    整理可得
    \begin{equation*}
        \boxed{\symbfit{a} \cdot \symbfit{b} + \symbfit{b} \cdot \symbfit{c} + \symbfit{c} \cdot \symbfit{a} = -\frac{3}{2}}
    \end{equation*}
\end{solution}

\begin{problem}{向量夹角}{Vector-Angle}
    若向量 $\symbfit{a} + 3\symbfit{b}$ 垂直于向量 $7\symbfit{a} - 5\symbfit{b}$，向量 $\symbfit{a} - 4\symbfit{b}$ 垂直于向量 $7\symbfit{a} - 2\symbfit{b}$，求两向量 $\symbfit{a}$ 和 $\symbfit{b}$ 间的夹角.
\end{problem}

\begin{solution}
    根据向量垂直的充要条件，内积为零，可得方程组
    \begin{equation*}
        \begin{cases}
            (\symbfit{a} + 3\symbfit{b}) \cdot (7\symbfit{a} - 5\symbfit{b}) = 0 \\
            (\symbfit{a} - 4\symbfit{b}) \cdot (7\symbfit{a} - 2\symbfit{b}) = 0
        \end{cases}
    \end{equation*}
    展开得
    \begin{equation*}
        \begin{cases}
            7|\symbfit{a}|^2 + 16\symbfit{a} \cdot \symbfit{b} - 15|\symbfit{b}|^2 = 0 \\
            7|\symbfit{a}|^2 - 30\symbfit{a} \cdot \symbfit{b} + 8|\symbfit{b}|^2 = 0
        \end{cases}
    \end{equation*}
    两式相减可消去 $|\symbfit{a}|^2$ 项
    \begin{equation*}
        46\symbfit{a} \cdot \symbfit{b} - 23|\symbfit{b}|^2 = 0 \implies \symbfit{a} \cdot \symbfit{b} = \frac{1}{2}|\symbfit{b}|^2
    \end{equation*}
    将上述关系代入第一式中
    \begin{equation*}
        7|\symbfit{a}|^2 + 16 \left( \frac{1}{2}|\symbfit{b}|^2 \right) - 15|\symbfit{b}|^2 = 0 \implies 7|\symbfit{a}|^2 - 7|\symbfit{b}|^2 = 0 \implies |\symbfit{a}| = |\symbfit{b}|
    \end{equation*}
    设向量 $\symbfit{a}$ 与 $\symbfit{b}$ 的夹角为 $\theta$，则有
    \begin{equation*}
        \cos \theta = \frac{\symbfit{a} \cdot \symbfit{b}}{|\symbfit{a}||\symbfit{b}|} = \frac{\frac{1}{2}|\symbfit{b}|^2}{|\symbfit{b}|^2} = \frac{1}{2}
    \end{equation*}
    因为 $\theta \in [0, \uppi]$，故
    \begin{equation*}
        \boxed{\theta = \frac{\uppi}{3}}
    \end{equation*}
\end{solution}

\begin{problem}{向量外积}{Vector-Cross-Product}
    已知向量 $\symbfit{a}$ 和 $\symbfit{b}$ 互相垂直，且 $|\symbfit{a}| = 3, |\symbfit{b}| = 4$，试计算：
    \begin{enumerate}
        \item $|(\symbfit{a} + \symbfit{b}) \times (\symbfit{a} - \symbfit{b})|$;
        \item $|(3\symbfit{a} - \symbfit{b}) \times (\symbfit{a} - 2\symbfit{b})|$.
    \end{enumerate}
\end{problem}

\begin{solution}
    \ansitem{1}

    利用外积的分配律与性质 $\symbfit{a} \times \symbfit{a} = \symbfit{0}, \symbfit{b} \times \symbfit{a} = - \symbfit{a} \times \symbfit{b}$ 展开
    \begin{equation*}
        \begin{aligned}
            (\symbfit{a} + \symbfit{b}) \times (\symbfit{a} - \symbfit{b}) & = \symbfit{a} \times \symbfit{a} - \symbfit{a} \times \symbfit{b} + \symbfit{b} \times \symbfit{a} - \symbfit{b} \times \symbfit{b} \\
                                                                           & = \symbfit{0} - \symbfit{a} \times \symbfit{b} - \symbfit{a} \times \symbfit{b} - \symbfit{0}                                       \\
                                                                           & = -2(\symbfit{a} \times \symbfit{b})
        \end{aligned}
    \end{equation*}
    由于 $\symbfit{a} \perp \symbfit{b}$，夹角为 $\uppi/2$，故其模长为
    \begin{equation*}
        |(\symbfit{a} + \symbfit{b}) \times (\symbfit{a} - \symbfit{b})| = 2|\symbfit{a}||\symbfit{b}|\sin \frac{\uppi}{2} = 2 \times 3 \times 4 \times 1
    \end{equation*}
    \begin{equation*}
        \boxed{|(\symbfit{a} + \symbfit{b}) \times (\symbfit{a} - \symbfit{b})| = 24}
    \end{equation*}

    \ansitem{2}

    同理展开被积表达式
    \begin{equation*}
        \begin{aligned}

\begin{problem}{向量恒等式证明}{Vector-Cross-Product-Identity}
    已知 $\symbfit{a} + \symbfit{b} + \symbfit{c} = \symbfit{0}$，试证
    \[ \symbfit{a} \times \symbfit{b} = \symbfit{b} \times \symbfit{c} = \symbfit{c} \times \symbfit{a}. \]
\end{problem}

\begin{myproof}
    由已知条件得 $\symbfit{a} + \symbfit{b} = -\symbfit{c}$。在此等式两端左叉乘 $\symbfit{a}$ 有
    \begin{equation*}
        \symbfit{a} \times (\symbfit{a} + \symbfit{b}) = \symbfit{a} \times (-\symbfit{c}) \implies \symbfit{a} \times \symbfit{a} + \symbfit{a} \times \symbfit{b} = -\symbfit{a} \times \symbfit{c}
    \end{equation*}
    由于 $\symbfit{a} \times \symbfit{a} = \symbfit{0}$ 且 $-\symbfit{a} \times \symbfit{c} = \symbfit{c} \times \symbfit{a}$，故
    \begin{equation*}
        \symbfit{a} \times \symbfit{b} = \symbfit{c} \times \symbfit{a}
    \end{equation*}
    同理，在等式 $\symbfit{b} + \symbfit{c} = -\symbfit{a}$ 两端左叉乘 $\symbfit{b}$ 得
    \begin{equation*}
        \symbfit{b} \times (\symbfit{b} + \symbfit{c}) = \symbfit{b} \times (-\symbfit{a}) \implies \symbfit{b} \times \symbfit{b} + \symbfit{b} \times \symbfit{c} = -\symbfit{b} \times \symbfit{a}
    \end{equation*}
    即
    \begin{equation*}
        \symbfit{b} \times \symbfit{c} = \symbfit{a} \times \symbfit{b}
    \end{equation*}
    综上所述，原命题成立。
\end{myproof}


\begin{problem}{非共线向量和}{Non-Collinear-Sum}
    已知 $\symbfit{a}, \symbfit{b}, \symbfit{c}$ 不共线，且 $\symbfit{a} \times \symbfit{b} = \symbfit{b} \times \symbfit{c} = \symbfit{c} \times \symbfit{a}$，求证：$\symbfit{a} + \symbfit{b} + \symbfit{c} = \symbfit{0}$。
\end{problem}

\begin{myproof}
    由 $\symbfit{a} \times \symbfit{b} = \symbfit{b} \times \symbfit{c}$ 可得
    \begin{equation*}
        \symbfit{a} \times \symbfit{b} - \symbfit{b} \times \symbfit{c} = \symbfit{0} \implies \symbfit{a} \times \symbfit{b} + \symbfit{c} \times \symbfit{b} = \symbfit{0} \implies (\symbfit{a} + \symbfit{c}) \times \symbfit{b} = \symbfit{0}
    \end{equation*}
    这意味着向量 $\symbfit{a} + \symbfit{c}$ 与 $\symbfit{b}$ 平行，设 $\symbfit{a} + \symbfit{c} = \lambda \symbfit{b}$。

    又由 $\symbfit{a} \times \symbfit{b} = \symbfit{c} \times \symbfit{a}$ 可得
    \begin{equation*}
        \symbfit{a} \times \symbfit{b} - \symbfit{c} \times \symbfit{a} = \symbfit{0} \implies \symbfit{a} \times \symbfit{b} + \symbfit{a} \times \symbfit{c} = \symbfit{0} \implies \symbfit{a} \times (\symbfit{b} + \symbfit{c}) = \symbfit{0}
    \end{equation*}
    这意味着向量 $\symbfit{b} + \symbfit{c}$ 与 $\symbfit{a}$ 平行，设 $\symbfit{b} + \symbfit{c} = \mu \symbfit{a}$。

    将 $\symbfit{c} = \mu \symbfit{a} - \symbfit{b}$ 代入第一个关系式 $\symbfit{a} + \symbfit{c} = \lambda \symbfit{b}$ 中
    \begin{equation*}
        \symbfit{a} + (\mu \symbfit{a} - \symbfit{b}) = \lambda \symbfit{b} \implies (1 + \mu) \symbfit{a} = (1 + \lambda) \symbfit{b}
    \end{equation*}
    由于 $\symbfit{a}, \symbfit{b}$ 不共线，故系数必全为零
    \begin{equation*}
        1 + \mu = 0, \quad 1 + \lambda = 0 \implies \mu = -1, \quad \lambda = -1
    \end{equation*}
    由此推得 $\symbfit{b} + \symbfit{c} = -\symbfit{a}$，即
    \begin{equation*}
        \symbfit{a} + \symbfit{b} + \symbfit{c} = \symbfit{0}
    \end{equation*}
\end{myproof}


\begin{problem}{拉格朗日恒等式}{Lagrange-Identity}
    求证：$|\symbfit{a} \times \symbfit{b}|^2 = |\symbfit{a}|^2 |\symbfit{b}|^2 - (\symbfit{a} \cdot \symbfit{b})^2$。

\begin{myproof}

    \end{equation*}
    于是：
    \begin{equation*}
        \begin{aligned}
            |\symbfit{a}|^2 |\symbfit{b}|^2 - (\symbfit{a} \cdot \symbfit{b})^2 & = |\symbfit{a}|^2 |\symbfit{b}|^2 - (|\symbfit{a}| |\symbfit{b}| \cos \theta)^2 \\
                                                                                & = |\symbfit{a}|^2 |\symbfit{b}|^2 (1 - \cos^2 \theta)                           \\
                                                                                & = |\symbfit{a}|^2 |\symbfit{b}|^2 \sin^2 \theta                                 \\
                                                                                & = (|\symbfit{a}| |\symbfit{b}| \sin \theta)^2                                   \\
                                                                                & = |\symbfit{a} \times \symbfit{b}|^2
        \end{aligned}
    \end{equation*}
    证毕。
\end{myproof}

        \begin{aligned}
             & \symbfit{a} \cdot \symbfit{b} = 5 \times 2 + 2 \times (-1) + 5 \times 2 = 10 - 2 + 10 = 18 \\
             & |\symbfit{b}| = \sqrt{2^2 + (-1)^2 + 2^2} = \sqrt{4 + 1 + 4} = 3
        \end{aligned}
    \end{equation*}
    由此得：
    \begin{equation*}
        \symbfit{a} \cdot \symbfit{e}_b = \frac{18}{3} = 6
    \end{equation*}
    \begin{equation*}
        \boxed{\symbfit{a} \cdot \symbfit{e}_b = 6}
    \end{equation*}
\end{solution}


\begin{problem}{向量积计算}{Vector-Product-Calculation-2}
    已知向量 $\symbfit{a} = (3, -1, -2)$ 和 $\symbfit{b} = (1, 2, -1)$，试求下列向量积的坐标：
    \begin{enumerate}
        \item $\symbfit{a} \times \symbfit{b}$；
        \item $(2\symbfit{a} - \symbfit{b}) \times (2\symbfit{a} + \symbfit{b})$。
    \end{enumerate}
\end{problem}

\begin{solution}
    \ansitem{1}

    根据向量积的行列式定义：
    \begin{equation*}
        \begin{aligned}
            \symbfit{a} \times \symbfit{b} & = \begin{vmatrix} \symbfit{i} & \symbfit{j} & \symbfit{k} \\ 3 & -1 & -2 \\ 1 & 2 & -1 \end{vmatrix}                                                                                   \\
                                           & = \symbfit{i} \begin{vmatrix} -1 & -2 \\ 2 & -1 \end{vmatrix} - \symbfit{j} \begin{vmatrix} 3 & -2 \\ 1 & -1 \end{vmatrix} + \symbfit{k} \begin{vmatrix} 3 & -1 \\ 1 & 2 \end{vmatrix} \\
                                           & = 5\symbfit{i} + \symbfit{j} + 7\symbfit{k}
        \end{aligned}
    \end{equation*}
    即
    \begin{equation*}
        \boxed{\symbfit{a} \times \symbfit{b} = (5, 1, 7)}
    \end{equation*}

    \ansitem{2}

    利用向量积的分配律及性质 $\symbfit{a} \times \symbfit{a} = \symbfit{0}$，$\symbfit{b} \times \symbfit{a} = -(\symbfit{a} \times \symbfit{b})$：
    \begin{equation*}
        \begin{aligned}
            (2\symbfit{a} - \symbfit{b}) \times (2\symbfit{a} + \symbfit{b}) & = 4(\symbfit{a} \times \symbfit{a}) + 2(\symbfit{a} \times \symbfit{b}) - 2(\symbfit{b} \times \symbfit{a}) - (\symbfit{b} \times \symbfit{b}) \\
                                                                             & = \symbfit{0} + 2(\symbfit{a} \times \symbfit{b}) + 2(\symbfit{a} \times \symbfit{b}) - \symbfit{0}                                            \\
                                                                             & = 4(\symbfit{a} \times \symbfit{b})
        \end{aligned}
    \end{equation*}
    代入第 (1) 问结果得：
    \begin{equation*}
        \boxed{(2\symbfit{a} - \symbfit{b}) \times (2\symbfit{a} + \symbfit{b}) = (20, 4, 28)}
    \end{equation*}
\end{solution}

\begin{problem}{三角形面积}{Triangle-Area}
    已知点 $A(1, 2, 0), B(3, 0, -3), C(5, 2, 6)$，求三角形 $ABC$ 的面积。
\end{problem}

\begin{solution}
    首先计算向量 $\overrightarrow{AB}$ 与 $\overrightarrow{AC}$ 的坐标：
    \begin{equation*}
        \overrightarrow{AB} = (2, -2, -3), \quad \overrightarrow{AC} = (4, 0, 6)
    \end{equation*}
    计算两向量的向量积：
    \begin{equation*}
        \begin{aligned}
            \overrightarrow{AB} \times \overrightarrow{AC} & = \begin{vmatrix} \symbfit{i} & \symbfit{j} & \symbfit{k} \\ 2 & -2 & -3 \\ 4 & 0 & 6 \end{vmatrix} \\
                                                           & = -12\symbfit{i} - (12 + 12)\symbfit{j} + 8\symbfit{k}                                              \\
                                                           & = (-12, -24, 8)
        \end{aligned}
    \end{equation*}
    三角形 $ABC$ 的面积 $S$ 为：
    \begin{equation*}
        \begin{aligned}
            S & = \frac{1}{2} \left| \overrightarrow{AB} \times \overrightarrow{AC} \right| \\
              & = \frac{1}{2} \sqrt{(-12)^2 + (-24)^2 + 8^2}                                \\
              & = \frac{1}{2} \sqrt{144 + 576 + 64} = \frac{1}{2} \sqrt{784} = \frac{28}{2}
        \end{aligned}
    \end{equation*}
    \begin{equation*}
        \boxed{S_{\triangle ABC} = 14}
    \end{equation*}
\end{solution}

\begin{problem}{四面体体积}{Tetrahedron-Volume}
    计算顶点为 $A(2, -1, 1), B(5, 5, 4), C(3, 2, -1), D(4, 1, 3)$ 的四面体的体积。
\end{problem}

\begin{solution}
    求出从顶点 $A$ 出发的三条棱对应的向量：
    \begin{equation*}
        \overrightarrow{AB} = (3, 6, 3), \quad \overrightarrow{AC} = (1, 3, -2), \quad \overrightarrow{AD} = (2, 2, 2)
    \end{equation*}
    计算三向量的混合积：
    \begin{equation*}
        \begin{aligned}
            (\overrightarrow{AB}, \overrightarrow{AC}, \overrightarrow{AD}) & = \begin{vmatrix} 3 & 6 & 3 \\ 1 & 3 & -2 \\ 2 & 2 & 2 \end{vmatrix} \\
                                                                            & = 3(6 + 4) - 6(2 + 4) + 3(2 - 6)                                     \\
                                                                            & = 30 - 36 - 12 = -18
        \end{aligned}
    \end{equation*}
    四面体的体积 $V$ 为混合积绝对值的 $\frac{1}{6}$：
    \begin{equation*}
        V = \frac{1}{6} \left| (\overrightarrow{AB}, \overrightarrow{AC}, \overrightarrow{AD}) \right| = \frac{1}{6} \cdot |-18| = 3
    \end{equation*}
    \begin{equation*}
        \boxed{V_{ABCD} = 3}
    \end{equation*}
\end{solution}


\begin{problem}{向量共面判断}{Vector-Coplanarity}
    判断下列向量 $\symbfit{a}, \symbfit{b}, \symbfit{c}$ 是否共面？
    \begin{enumerate}
        \item $\symbfit{a} = (2, 3, -1)$, $\symbfit{b} = (1, -1, 3)$, $\symbfit{c} = (1, 9, -11)$；
        \item $\symbfit{a} = (3, -2, 1)$, $\symbfit{b} = (2, 1, 2)$, $\symbfit{c} = (3, -1, -2)$。
    \end{enumerate}
\end{problem}

\begin{solution}
    \ansitem{1}
    计算向量 $\symbfit{a}, \symbfit{b}, \symbfit{c}$ 的混合积：
    \begin{equation*}
        (\symbfit{a}, \symbfit{b}, \symbfit{c}) = \begin{vmatrix} 2 & 3 & -1 \\ 1 & -1 & 3 \\ 1 & 9 & -11 \end{vmatrix} = 2(11 - 27) - 3(-11 - 3) - 1(9 + 1) = -32 + 42 - 10 = 0
    \end{equation*}
    因为混合积为零，故向量 $\symbfit{a}, \symbfit{b}, \symbfit{c}$ 共面。
    \begin{equation*}
        \boxed{\text{向量 } \symbfit{a}, \symbfit{b}, \symbfit{c} \text{ 共面}}
    \end{equation*}

    \ansitem{2}
    计算向量 $\symbfit{a}, \symbfit{b}, \symbfit{c}$ 的混合积：
    \begin{equation*}
        (\symbfit{a}, \symbfit{b}, \symbfit{c}) = \begin{vmatrix} 3 & -2 & 1 \\ 2 & 1 & 2 \\ 3 & -1 & -2 \end{vmatrix} = 3(-2 + 2) + 2(-4 - 6) + 1(-2 - 3) = 0 - 20 - 5 = -25 \neq 0
    \end{equation*}
    因为混合积不为零，故向量 $\symbfit{a}, \symbfit{b}, \symbfit{c}$ 不共面。
    \begin{equation*}
        \boxed{\text{向量 } \symbfit{a}, \symbfit{b}, \symbfit{c} \text{ 不共面}}
    \end{equation*}
\end{solution}

\begin{problem}{四点共面判断}{Four-Points-Coplanar}
    下列四点：$A(1, 2, -1), B(0, 1, 5), C(-1, 2, 1), D(2, 1, 3)$ 是否在一个平面上？
\end{problem}

\begin{solution}
    构造以 $A$ 点为起点的三个向量：
    \begin{equation*}
        \begin{aligned}
            \overrightarrow{AB} & = (0-1, 1-2, 5-(-1)) = (-1, -1, 6) \\
            \overrightarrow{AC} & = (-1-1, 2-2, 1-(-1)) = (-2, 0, 2) \\
            \overrightarrow{AD} & = (2-1, 1-2, 3-(-1)) = (1, -1, 4)
        \end{aligned}
    \end{equation*}
    计算这三个向量的混合积：
    \begin{equation*}
        (\overrightarrow{AB}, \overrightarrow{AC}, \overrightarrow{AD}) = \begin{vmatrix} -1 & -1 & 6 \\ -2 & 0 & 2 \\ 1 & -1 & 4 \end{vmatrix} = -(-2)(-4 + 6) + 0 - 2(1 + 1) = 4 - 4 = 0
    \end{equation*}
    由于混合积为零，表明向量 $\overrightarrow{AB}, \overrightarrow{AC}, \overrightarrow{AD}$ 共面。
    \begin{equation*}
        \boxed{A, B, C, D \text{ 四点共面}}
    \end{equation*}
\end{solution}

\begin{problem}{坐标对称性}{Coordinate-Symmetry}
    \begin{enumerate}
        \item 给定点 $A(a, b, c)$，求它关于三个坐标平面对称点的坐标；
        \item 求定点 $(a, b, c)$ 关于三个坐标轴对称点的坐标。
    \end{enumerate}
\end{problem}

\begin{solution}
    \ansitem{1}
    根据空间解析几何中的对称规则，点 $A(a, b, c)$ 关于各坐标平面的对称点分别为：
    \begin{itemize}
        \item 关于 $xOy$ 面：保持 $x, y$ 不变，$z$ 取相反数，即 $(a, b, -c)$；
        \item 关于 $yOz$ 面：保持 $y, z$ 不变，$x$ 取相反数，即 $(-a, b, c)$；
        \item 关于 $zOx$ 面：保持 $z, x$ 不变，$y$ 取相反数，即 $(a, -b, c)$。
    \end{itemize}
    \begin{equation*}
        \boxed{\text{关于 } xOy, yOz, zOx \text{ 面对称点分别为 } (a, b, -c), (-a, b, c), (a, -b, c)}
    \end{equation*}

    \ansitem{2}
    点 $(a, b, c)$ 关于各坐标轴的对称点分别为：
    \begin{itemize}
        \item 关于 $x$ 轴：保持 $x$ 不变，$y, z$ 取相反数，即 $(a, -b, -c)$；
        \item 关于 $y$ 轴：保持 $y$ 不变，$x, z$ 取相反数，即 $(-a, b, -c)$；
        \item 关于 $z$ 轴：保持 $z$ 不变，$x, y$ 取相反数，即 $(-a, -b, c)$。
    \end{itemize}
    \begin{equation*}
        \boxed{\text{关于 } x, y, z \text{ 轴对称点分别为 } (a, -b, -c), (-a, b, -c), (-a, -b, c)}
    \end{equation*}
\end{solution}


\begin{problem}{点到原点及坐标轴的距离}{Point-To-Axis-Distance}
    求点 $P(4, -3, 5)$ 到坐标原点以及到各坐标轴的距离。
\end{problem}

\begin{solution}
    设点 $P$ 的坐标为 $(x, y, z) = (4, -3, 5)$。根据空间两点间距离公式及点到坐标轴的距离公式，有
    \begin{equation*}
        \begin{aligned}
            d_{PO} & = \sqrt{x^2 + y^2 + z^2} = \sqrt{4^2 + (-3)^2 + 5^2} = \sqrt{50} = 5\sqrt{2} \\
            d_{Px} & = \sqrt{y^2 + z^2} = \sqrt{(-3)^2 + 5^2} = \sqrt{34}                         \\
            d_{Py} & = \sqrt{x^2 + z^2} = \sqrt{4^2 + 5^2} = \sqrt{41}                            \\
            d_{Pz} & = \sqrt{x^2 + y^2} = \sqrt{4^2 + (-3)^2} = \sqrt{25} = 5
        \end{aligned}
    \end{equation*}
    因此，点 $P$ 到原点、 $x$ 轴、 $y$ 轴、 $z$ 轴的距离分别为
    \begin{equation*}
        \boxed{d_{PO} = 5\sqrt{2}, \quad d_{Px} = \sqrt{34}, \quad d_{Py} = \sqrt{41}, \quad d_{Pz} = 5}
    \end{equation*}
\end{solution}

\begin{problem}{平面内的等距点}{Equidistant-Point-In-Plane}
    在 $yz$ 平面上求一点 $P$，使它与三已知点 $A(3, 1, 2)$, $B(4, -2, -2)$ 及 $C(0, 5, 1)$ 等距离。
\end{problem}

\begin{solution}
    因为点 $P$ 在 $yz$ 平面上，设其坐标为 $P(0, y, z)$。由题意 $PA = PB = PC$，即 $PA^2 = PB^2$ 且 $PA^2 = PC^2$。
    \begin{equation*}
        \begin{cases}
            (0 - 3)^2 + (y - 1)^2 + (z - 2)^2 = (0 - 4)^2 + (y + 2)^2 + (z + 2)^2 \\
            (0 - 3)^2 + (y - 1)^2 + (z - 2)^2 = (0 - 0)^2 + (y - 5)^2 + (z - 1)^2
        \end{cases}
    \end{equation*}
    展开并整理各项得
    \begin{equation*}
        \begin{cases}
            y^2 + z^2 - 2y - 4z + 14 = y^2 + z^2 + 4y + 4z + 24 \\
            y^2 + z^2 - 2y - 4z + 14 = y^2 + z^2 - 10y - 2z + 26
        \end{cases}
    \end{equation*}
    \begin{equation*}
        \implies
        \begin{cases}
            6y + 8z = -10 \\
            8y - 2z = 12
        \end{cases}
        \iff
        \begin{cases}
            3y + 4z = -5 \\
            4y - z = 6
        \end{cases}
    \end{equation*}
    由第二个方程得 $z = 4y - 6$，代入第一个方程
    \begin{equation*}
        3y + 4(4y - 6) = -5 \implies 19y = 19 \implies y = 1
    \end{equation*}
    进而求得 $z = 4(1) - 6 = -2$。故点 $P$ 的坐标为
    \begin{equation*}
        \boxed{P(0, 1, -2)}
    \end{equation*}
\end{solution}
